# 手写节流

## 手写节流函数的尾部执行控制。

```html
<script>
  function ztThrottle(fn, interval, { leading = true, trailing = false } = {}) {
    let startTime = 0
    let timer = null
    const _throttle = function(...args) {
      const nowTime = new Date().getTime() // 1.获取当前时间
      if (!leading && startTime === 0) { // 对立即执行进行控制
        startTime = nowTime
      }
      const waitTime = interval - (nowTime - startTime) // 2.计算需要等待的时间执行函数
      if (waitTime <= 0) {
        if (timer) clearTimeout(timer)
        fn.apply(this, args)
        startTime = nowTime
        timer = null
        return
      }
      if (trailing && !timer) { // 3.判断是否需要执行尾部
        timer = setTimeout(() => {
          fn.apply(this, args)
          startTime = leading ? new Date().getTime() : 0
          timer = null
        }, waitTime);
      }
    }
    return _throttle
  }
</script>
<script>
  // 1.获取input元素
  const inputEl = document.querySelector("input")
  // 3.自己实现的节流函数
  let counter = 1
  inputEl.oninput = ztthrottle(function(event) {
    console.log(`发送网络请求${counter++}:`, this.value, event)
  }, 3000, { trailing: true })
</script>
```

## 手写节流函数的取消功能和返回值功能。

```html
<script>
  function ztthrottle(fn, interval, { leading = true, trailing = false } = {}) {
    let startTime = 0
    let timer = null
    const _throttle = function(...args) {
      return new Promise((resolve, reject) => {
        try {
          const nowTime = new Date().getTime() // 1.获取当前时间
          if (!leading && startTime === 0) { // 对立即执行进行控制
            startTime = nowTime
          }
          const waitTime = interval - (nowTime - startTime) // 2.计算需要等待的时间执行函数
          if (waitTime <= 0) {
            if (timer) clearTimeout(timer)
            const res = fn.apply(this, args)
            resolve(res)
            startTime = nowTime
            timer = null
            return
          }
          if (trailing && !timer) { // 3.判断是否需要执行尾部
            timer = setTimeout(() => {
              const res = fn.apply(this, args)
              resolve(res)
              startTime = leading ? new Date().getTime() : 0
              timer = null
            }, waitTime);
          }
        } catch (error) {
          reject(error)
        }
      })
    }
    _throttle.cancel = function() {
      if (timer) clearTimeout(timer)
      startTime = 0
      timer = null
    }
    return _throttle
  }
</script>
<script>
  // 1.获取input元素
  const inputEl = document.querySelector("input")
  const cancelBtn = document.querySelector(".cancel")
  // 3.自己实现的节流函数
  let counter = 1
  const throttleFn = ztthrottle(function(event) {
    console.log(`发送网络请求${counter++}:`, this.value, event)
    return "throttle return value"
  }, 3000, { trailing: true })
  inputEl.oninput = function (...args) {
    throttleFn.apply(this, args).then(res => {
      console.log("res:", res)
    })
  }
</script>
```

# 手写深拷贝

总结对象相互赋值的一些关系3点

- 引入的赋值：指向同一个对象，相互之间会影响；
- 对象的浅拷贝：只是浅层的拷贝，内部引入对象时，依然会相互影响；
- 对象的深拷贝：两个对象不再有任何关系，不会相互影响；

-----

为什么JS没有提供深拷贝的方法：会消耗大量内存。

-----

使用JSON实现深拷贝有什么弊端。

- 这种深拷贝的方式其实对于**函数**、**Symbol**等是无法处理的； 
- 并且如果存在**对象的循环引用**，也会报错的；

-----

## 手写深拷贝的基本实现。

```html
<script>
	function isObject(value) {
		const valueType = typeof value
		return (value !== null) && ( valueType === "object" || valueType === "function" )
	}
	function deepCopy(originValue) {
		if (!isObject(originValue)) { // 1.如果是原始类型, 直接返回
			return originValue
		}
		const newObj = {} // 2.如果是对象类型, 才需要创建对象
		Object.keys(originValue).forEach(keys => {
			newObj[key] = deepCopy(originValue[key]);
		})
		return newObj
	}
</script>
<script>
	const info = {
		name: "zzt",
		age: 18,
		friend: {
			name: "kobe",
			address: {
				name: "洛杉矶",
				detail: "斯坦普斯中心"
			}
		}
	}
	const newObj = deepCopy(info)
	info.friend.address.name = "北京市"
	console.log(newObj.friend.address.name)
</script>
```

## 手写深拷贝函数的其它类型处理

```html
<script>
	// 深拷贝函数
	function deepCopy(originValue) {
		if (typeof originValue === "symbol") { // 0.如果值是Symbol的类型
			return Symbol(originValue.description)
		}
		if (!isObject(originValue)) { // 1.如果是原始类型, 直接返回
			return originValue
		}
    if (typeof originValue === "function") { // 3.如果是函数function类型, 不需要进行深拷贝
			return originValue
		}
		const newObj = null
		if (originValue instanceof Set) { // 2.1.如果是set类型
			newObj = new Set()
			originValue.forEach(item => {
				newObj.add(deepCopy(item))
			})
			return newObj
		}
		if (originValue instanceof Map) { // 2.2.如果是Map类型
			newObj = new Map()
			originValue.forEach((value, key) => {
				newObj.set(deepCopy(key), deepCopy(value))
			});
			return newObj
		}
		if (Array.isArray(originValue)) {
			newObj = []
			originValue.forEach((item, index) => {
				newObj[index] = deepCopy(item);
			})
		} else {
			newObj = {}
			Object.keys(originValue).forEach(keys => {
				newObj[key] = deepCopy(originValue[key]);
			})
			// 单独遍历 Symbol 类型的 key
			Object.getOwnPropertySymbols(originValue).forEach(skey => {
				newObj[Symbol(skey.description)] = deepCopy(originValue[skey])
			})
		}
		return newObj
	}
</script>
<script>
	const set = new Set(["abc", "cba", "nba"])
	const s1 = Symbol("s1")
	const s2 = Symbol("s2")
	const info = {
		name: "zzt",
		age: 18,
		friend: {
			name: "kobe",
			address: {
				name: "洛杉矶",
				detail: "斯坦普斯中心"
			}
		},
		// 1.特殊类型: Set
		set: set,
		// 2.特性类型: function
		running: function() {
			console.log("running~")
		},
		// 3.值的特殊类型: Symbol
		symbolKey: Symbol("abc"),
		// 4.key是symbol时
		[s1]: "aaaa",
		[s2]: "bbbb"
	}
	const newObj = deepCopy(info)
	console.log(newObj)
</script>
```

## 手写深拷贝循环引用的处理。

```html
<script>
	// let map = new WeakMap()
	function deepCopy(originValue, map = new WeakMap()) {
		// const map = new WeakMap()
		if (typeof originValue === "symbol") { // 0.如果值是Symbol的类型
			return Symbol(originValue.description)
		}
		if (!isObject(originValue)) { // 1.如果是原始类型, 直接返回
			return originValue
		}
    if (typeof originValue === "function") { // 3.如果是函数function类型, 不需要进行深拷贝
			return originValue
		}
    if (map.has(originValue)) { // 4.如果是对象类型, 才需要创建对象
			return map.get(originValue)
		}
		let newObj = null
		if (originValue instanceof Set) { // 2.1.如果是set类型
			newObj = new Set()
			map.set(originValue, newObj)
			originValue.forEach(item => {
				newObj.add(deepCopy(item, map))
			})
			return newObj
		}
		if (originValue instanceof Map) { // 2.2.如果是Map类型
			newObj = new Map()
			map.set(originValue, newObj)
			originValue.forEach((value, key) => {
				newObj.set(deepCopy(key, map), deepCopy(value, map))
			});
			return newObj
		}
		if (Array.isArray(originValue)) {
			newObj = []
			map.set(originValue, newObj)
			originValue.forEach((item, index) => {
				newObj[index] = deepCopy(item, map);
			})
		} else {
			newObj = {}
			map.set(originValue, newObj)
			Object.keys(originValue).forEach(keys => {
				newObj[key] = deepCopy(originValue[key], map);
			})
			// 单独遍历symbol
			Object.getOwnPropertySymbols(originValue).forEach(skey => {
				newObj[Symbol(skey.description)] = deepCopy(originValue[skey], map)
			});
		}
		return newObj
	}
</script>

		name: "zzt",
		age: 18,
		friend: {
			name: "kobe",
			address: {
				name: "洛杉矶",
				detail: "斯坦普斯中心"

		},
		self: info
	}

	console.log(newObj.self === newObj)
```

# 事件总线

事件总线通常用来做跨文件/组件的通信，什么是事件总线？

- 事件总线属于一种观察者模式，其中包括三个角色： 
	- 发布者（Publisher）：发出事件（Event）； 
	- 订阅者（Subscriber）：订阅事件（Event），并且会进行响应（Handler）；
	- 事件总线（EventBus）：无论是发布者还是订阅者都是通过事件总线作为中台的；
-----

有哪些第三方库实现了事件总线。

- Vue2默认是带有事件总线的功能； 
- Vue3中推荐一些第三方库，比如 `mitt` 或 `tiny-emitter`；

-----

事件总线的使用过程，伪代码实现。

```JavaScript
// 使用过程
const eventBus = new ZTEventBus()
// aside.vue 组件中监听事件
eventBus.on("navclick", (name, age, height) => {
	console.log("navclick listener 01", name, age, height)
})
const click =  () => {
	console.log("navclick listener 02")
}
eventBus.on("navclick", click)
setTimeout(() => {
	eventBus.off("navclick", click)
}, 5000);
eventBus.on("asideclick", () => {
	console.log("asideclick listener")
})
// nav.vue 组件中发布事件
const navBtnEl = document.querySelector(".nav-btn")
navBtnEl.onclick = function() {
	console.log("自己监听到")
	eventBus.emit("navclick", "zzt", 18, 1.88)
}
```

事件总线核心逻辑实现。

```javascript
// 类EventBus -> 事件总线对象
class ZTEventBus {
	constructor() {
		this.eventbus = {}
	}
	on(eventName, eventFn, thisArg) {
		let handlers = this.eventbus[eventName]
		if (!handlers) {
			handlers = []
			this.eventbus[eventName] = handlers
		}
		handlers.push({eventFn, thisArg})
	}
	off(eventName, eventFn) {
		let handlers = this.eventbus[eventName]
		if (!handlers) return
		const index = handlers.findIndex(item => item.eventFn === eventFn)
		handlers.splice(index, 1)
		// 如果eventFns已经清空了
		if (handlers.length === 0) {
			delete this.eventbus[eventName]
		}
	}
	emit(eventName, ...args) {
		let handlers = this.eventbus[eventName]
		if (!handlers) return
		handlers.forEach(item => {
			item.eventFn.apply(item.thisArg, args)
		})
	}
}
```
